76 private async Task LoadAllAsync(int id)
77 {
78 // load statuses
79 Statuses.Clear();
80 foreach (var s in await _ticketService.GetAllTicketStatusesAsync())
81 Statuses.Add(s);
82
83 // load ticket and select its current status
84 Ticket = await _ticketService.GetTicketByIdAsync(id);
85 if (Ticket != null)
86 SelectedStatus = Statuses.First(s => s.Id == Ticket.StatusId);
87
88 // load response log
89 Responses.Clear();
90 foreach (var r in await _ticketService.GetTicketResponsesAsync(id))
91 Responses.Add(r);

98 [RelayCommand]
99 private async Task SaveAsync()
100 {
101 if (Ticket == null) return;
102
103 // 1) Did status change?
104 if (Ticket.StatusId != SelectedStatus.Id)
105 {
106 var ok = await _ticketService.ChangeTicketStatusAsync(
107 Ticket.Id, SelectedStatus.Id);
108 if (!ok)
109 {
110 await Shell.Current.DisplayAlert("Error",
111 "Failed to update status", "OK");
112 return;
113 }
114 Ticket.StatusId = SelectedStatus.Id;
115 }
116
117 // 2) Always log a response (even if Note blank)
118 var resp = new TicketResponse
119 {
120 TicketId = Ticket.Id,
121 StatusId = SelectedStatus.Id,
122 Note = Note.Trim(),
123 CreatedAt = DateTime.UtcNow
124 };
125 if (! await _ticketService.AddTicketResponseAsync(resp))
126 {
127 await Shell.Current.DisplayAlert("Error",
128 "Failed to log response", "OK");
129 return;
130 }
131
132 // 3) reload everything
133 Note = string.Empty;
134 await LoadAllAsync(Ticket.Id);
135 await Shell.Current.DisplayAlert("Saved",
136 "Status & response recorded.", "OK");
137 }
