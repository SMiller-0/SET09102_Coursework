103 [RelayCommand]
104 private async Task SaveSensor()
105 {
106 HasError = false;
107 ErrorMessage = string.Empty;
108
109 if (AreRequiredFieldsMissing())
110 {
111 ErrorMessage = "Please fill in all required fields.";
112 HasError = true;
113 return;
114 }
115
116 if (!decimal.TryParse(LatitudeInput, out var latitude))
117 {
118 ErrorMessage = "Latitude must be a valid number.";
119 HasError = true;
120 return;
121 }
122
123 if (!decimal.TryParse(LongitudeInput, out var longitude))
124 {
125 ErrorMessage = "Longitude must be a valid number.";
126 HasError = true;
127 return;
128 }
129
130 var newSensor = new Sensor
131 {
132 Name = Name.Trim(),
133 FirmwareVersion = FirmwareVersion.Trim(),
134 SensorTypeId = SelectedSensorType.Id,
135 Longitude = longitude,
136 Latitude = latitude,
137 IsActive = IsActive
138 };
139
140 var validationResult = _sensorValidator.Validate(newSensor);
141 if (!validationResult.IsValid)
142 {
143 ErrorMessage = validationResult.ErrorMessage;
144 HasError = true;
145 return;
146 }
147
148 var success = await _sensorService.AddSensorAsync(newSensor);
149
150 if (success)
151 {
152 await Shell.Current.DisplayAlert("Success", "Sensor added successfully!", "OK");
153 await _navigationService.NavigateToAllSensorsAsync();
154 }
155 else
156 {
157 ErrorMessage = "An error occurred while saving the sensor.";
158 HasError = true;
159 }
