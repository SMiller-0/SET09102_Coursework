116 [RelayCommand]
117 private async Task Create()
118 {
119 // Validation checks:
120 if (string.IsNullOrWhiteSpace(FirstName)
121 || string.IsNullOrWhiteSpace(Surname)
122 || string.IsNullOrWhiteSpace(Email)
123 || string.IsNullOrWhiteSpace(Password)
124 || Password != ConfirmPassword
125 || SelectedRole == null)
126 {
127 await Shell.Current.DisplayAlert("Missing Info",
128 "Please fill out all fields and ensure password matches.","OK");
129 return;
130 }
131
132 var emailPattern = @"^[^@\s]+@[^@\s]+\.[^@\s]+$";
133 if (!Regex.IsMatch(Email, emailPattern))
134 {
135 await Shell.Current.DisplayAlert(
136 "Invalid Email", "Please enter a valid email address.","OK");
137 return;
138 }
139
140 var postcodePattern = @"^[A-Z0-9 ]{3,20}$";
141 if (!Regex.IsMatch(Postcode.ToUpper(), postcodePattern))
142 {
143 await Shell.Current.DisplayAlert(
144 "Invalid Postcode","Please enter a valid postcode.","OK");
145 return;
146 }
147
148 if (!string.IsNullOrWhiteSpace(PhoneNumber))
149 {
150 var phonePattern = @"^[\d\+\-\s]{5,20}$";
151 if (!Regex.IsMatch(PhoneNumber, phonePattern))
152 {
153 await Shell.Current.DisplayAlert(
154 "Invalid Phone","Please enter a valid phone number.","OK");
155 return;
156 }
157 }
158
159 var pwdPattern = @"^(?=.*[a-z])(?=.*[A-Z])(?=.*\d).{8,}$";
160 if (!Regex.IsMatch(Password, pwdPattern))
161 {
162 await Shell.Current.DisplayAlert(
163 "Weak Password",
164 "Password must be at least 8 characters and include uppercase, lowercase, and a number.",
165 "OK");
166 return;
167 }
168
169 // Hash password with BCrypt and create user entity
170 var hashed = BCrypt.Net.BCrypt.HashPassword(Password);
171
172 var user = new User
173 {
174 FirstName = FirstName,
175 MiddleName = String.IsNullOrWhiteSpace(MiddleName) ? null : MiddleName,
176 Surname = Surname,
177 Email = Email,
178 PhoneNumber = String.IsNullOrWhiteSpace(phoneNumber) ? null : PhoneNumber,
179 Street = Street,
180 City = City,
181 Postcode = Postcode,
182 Password = hashed,
183 Role = SelectedRole
184 };
185
186 _context.Users.Add(user);
187
188 try
189 {
190 // Attempt to save to DB
191 await _context.SaveChangesAsync();
192 }
193 catch (DbUpdateException dbEx)
194 {
195 // handle duplicate‐email (UNIQUE index) errors
196 if (dbEx.InnerException?.Message.Contains("IX_Users_Email") ?? false)
197 {
198 await Shell.Current.DisplayAlert(
199 "Duplicate Email",
200 "That email is already in use. Please choose another.",
201 "OK");
202 return;
203 }
204
205 // Any other database error
206 await Shell.Current.DisplayAlert(
207 "Database Error",
208 "Unable to save user right now. Please try again later.",
209 "OK");
210 return;
211 }
212 catch (Exception ex)
213 {
214 // Any other error
215 await Shell.Current.DisplayAlert(
216 "Unexpected Error",
217 $"Something went wrong:\n{ex.Message}",
218 "OK");
219 return;
220 }
221
222 await Shell.Current.DisplayAlert("Success", "New user created.", "OK");
223 await Shell.Current.GoToAsync($"///AllUsersPage?created={user.Id}");
224 }
