118 [RelayCommand]
119 private async Task Save()
120 {
121 HasError = false;
122 ErrorMessage = string.Empty;
123
124 if (string.IsNullOrWhiteSpace(Name) || string.IsNullOrWhiteSpace(FirmwareVersion) ||
125 string.IsNullOrWhiteSpace(LatitudeInput) || string.IsNullOrWhiteSpace(LongitudeInput) ||
126 SelectedSensorType == null)
127 {
128 ErrorMessage = "Please fill in all required fields.";
129 HasError = true;
130 return;
131 }
132
133 if (!decimal.TryParse(LatitudeInput, out var latitude))
134 {
135 ErrorMessage = "Latitude must be a valid number.";
136 HasError = true;
137 return;
138 }
139
140 if (!decimal.TryParse(LongitudeInput, out var longitude))
141 {
142 ErrorMessage = "Longitude must be a valid number.";
143 HasError = true;
144 return;
145 }
146
147 Sensor.Name = Name.Trim();
148 Sensor.FirmwareVersion = FirmwareVersion.Trim();
149 Sensor.Latitude = latitude;
150 Sensor.Longitude = longitude;
151 Sensor.SensorTypeId = SelectedSensorType.Id;
152 Sensor.IsActive = IsActive;
153
154 var validationResult = _sensorValidator.Validate(Sensor);
155 if (!validationResult.IsValid)
156 {
157 ErrorMessage = validationResult.ErrorMessage;
158 HasError = true;
159 return;
160 }
161
162 try
163 {
164 var success = await _sensorService.UpdateSensorAsync(Sensor);
165
166 if (success)
167 {
168 await Shell.Current.DisplayAlert("Success", "Sensor updated successfully!", "OK");
169 await _navigationService.NavigateToAllSensorsAsync(refresh: true);
170 }
171 else
172 {
173 ErrorMessage = "Failed to update sensor.";
174 HasError = true;
175 }
176 }
177 catch (Exception ex)
178 {
179 ErrorMessage = $"Unexpected error: {ex.Message}";
180 HasError = true;
